• dump 함수
    ・ 인스턴스의 자세한 설명(description 프로퍼티)까지 출력

    class Person {
        var name: String = "Yoonseo"
        var age: Int = 18
    }

    let Yoonseo: Person = Person() 

    print(Yoonseo) //Person의 인스턴스라고만 표현한다.

    dump(Yoonseo) // Person의 인스턴스라고 표현
                     name: "Yoonseo"
                     age: 18
